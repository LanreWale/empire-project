  /* ======= UTILS ======= */
  const $ = (id) => document.getElementById(id);
  const esc = s => String(s ?? '').replace(/[&<>"']/g, m => ({'&':'&','<':'<','>':'>','"':'"',"'":'''}[m]));
  const num = n => (+n||0).toLocaleString();
  const usd = n => new Intl.NumberFormat('en-US',{style:'currency',currency:'USD'}).format(+n||0);
  async function jget(url){
    const r = await fetch(url, {cache:'no-store', headers:{'Accept':'application/json'}});
    if(!r.ok) throw new Error(`${r.status} ${r.statusText}`);
    return r.json();
  }
  function showError(el, msg){ el.classList.remove('hidden'); el.textContent = msg; }

  /* ======= TABS ======= */
  function showTab(key){
    document.querySelectorAll('.nav button').forEach(b=>{
      b.classList.toggle('active', b.dataset.tab===key);
    });
    document.querySelectorAll('.view').forEach(v=>v.classList.add('hidden'));
    const view = document.getElementById('view-'+key);
    if(view) view.classList.remove('hidden');

    if(key==='dashboard') loadOverview();
    if(key==='cpa')       loadCpa();
    if(key==='users')     loadUsers();
    if(key==='analytics') loadAnalytics();
    if(key==='wallet')    loadWallet();
    if(key==='monitoring')loadMonitoring();
    if(key==='settings')  loadSettings();
  }
  document.querySelectorAll('.nav button').forEach(b=>{
    b.addEventListener('click',()=>{ location.hash=b.dataset.tab; showTab(b.dataset.tab); });
  });

  /* ======= OVERVIEW ======= */
  async function loadOverview(){
    try{
      const j = await jget(api("overview"));
      const t = j.totals || {};
      const tp = j.totalsPretty || {};

      $('kpi-earn').textContent  = usd(t.earnings || 0);
      $('kpi-leads').textContent = num(t.leads || 0);
      $('kpi-clicks').textContent= num(t.clicks || 0);
      $('kpi-epc').textContent   = usd(t.epc || 0);
      $('kpi-cpa').textContent   = usd(t.cpa || 0);
      $('kpi-rpm').textContent   = usd(t.rpm || 0);
      $('kpi-cr').textContent    = tp.conversionRatePct || ((t.conversionRate ?? 0).toFixed(2) + '%');

      const fillObj = (tbody, obj) => {
        tbody.innerHTML = '';
        Object.entries(obj||{}).forEach(([k,v])=>{
          const tr=document.createElement('tr');
          tr.innerHTML = `${esc(k)}${usd(v)}`;
          tbody.appendChild(tr);
        });
        if(!tbody.children.length){ tbody.innerHTML = `No data`; }
      };
      fillObj($('#tbl-geo tbody'), j.geo);
      fillObj($('#tbl-device tbody'), j.devices);
      fillObj($('#tbl-offer tbody'), j.offerTypes);

      const dayBody = $('#tbl-day tbody'); dayBody.innerHTML='';
      const byDay = Array.isArray(j.byDay) ? j.byDay
                   : Object.entries(j.byDay||{}).map(([d,v])=>({t:d, earnings:v}));
      byDay.forEach(r=>{
        const tr=document.createElement('tr');
        tr.innerHTML=`${esc(r.t||r.date||'')}${usd(r.earnings||r.value||0)}`;
        dayBody.appendChild(tr);
      });
      if(!dayBody.children.length) dayBody.innerHTML=`No data`;

      $('ovrErr').classList.add('hidden');
    }catch(e){
      showError($('ovrErr'), `Failed to load overview: ${e.message}`);
      console.warn('overview failed', e);
    }
  }

  /* ======= CPA ======= */
  async function loadCpa(){
    const grid=$('cpaGrid'); grid.innerHTML='';
    try{
      const j = await jget(api("cpa"));
      (j.items||[]).forEach(acc=>{
        const name = acc.account || acc.name || 'CPA';
        const network = acc.network || '—';
        const offers = acc.activeOffers ?? acc.offers ?? 0;
        const revenue = acc.revenue ?? 0;
        const clicks = acc.clicks ?? 0;
        const conv = (()=>{
          if (acc.conversionRate != null) {
            const n = (typeof acc.conversionRate === 'string' && acc.conversionRate.endsWith('%'))
              ? acc.conversionRate
              : ((+acc.conversionRate)*100).toFixed(1)+'%';
            return n;
          }
          if (acc.conversion != null) {
            return (typeof acc.conversion === 'string' && acc.conversion.endsWith('%'))
              ? acc.conversion
              : ((+acc.conversion)*100).toFixed(1)+'%';
          }
          return '—';
        })();

        const card=document.createElement('div'); card.className='card span-4';
        card.innerHTML = `
          
${esc(name)}

          
Network
${esc(network)}

          
Active Offers
${num(offers)}

          
Revenue
${usd(revenue)}

          
Clicks
${num(clicks)}

          
Conversion
${conv}
`;
        grid.appendChild(card);
      });
      if(!grid.children.length) grid.innerHTML=`
No CPA accounts returned.
`;
    }catch(e){
      grid.innerHTML=`
Failed to load CPA: ${esc(e.message)}
`;
    }
  }

  /* ======= USERS ======= */
  async function loadUsers(){
    const body=$('#tbl-users tbody'); body.innerHTML='';
    try{
      const j = await jget(api("users"));
      (j.items||[]).forEach(u=>{
        const emailClean = String(u.email||'').replace(/\+$/,'').trim();
        const tr=document.createElement('tr');
        tr.innerHTML=`${esc(u.name||'')}${esc(emailClean)}${esc(u.phone||'')}`;
        body.appendChild(tr);
      });
      if(!body.children.length) body.innerHTML=`No users.`;
    }catch(e){
      body.innerHTML=`Failed to load users.`;
    }
  }

  /* ======= ANALYTICS ======= */
  async function loadAnalytics(){
    const body=$('#tbl-analytics tbody'); body.innerHTML='';
    try{
      const j = await jget(api("analytics"));
      // Supports {items:[{metric,value}]} OR {totals:{}, ...} OR passthrough row list
      if (Array.isArray(j.items) && j.items.length && j.items[0].metric !== undefined) {
        j.items.forEach(r=>{
          const tr=document.createElement('tr');
          tr.innerHTML=`${esc(r.metric||'')}${esc(r.value||'')}`;
          body.appendChild(tr);
        });
      } else if (j.totals) {
        Object.entries(j.totals).forEach(([k,v])=>{
          const tr=document.createElement('tr');
          tr.innerHTML=`${esc(k)}${esc(v)}`;
          body.appendChild(tr);
        });
      } else if (Array.isArray(j.items)) {
        // Generic rows: show first 6 key/value pairs per row
        j.items.forEach(r=>{
          const first = Object.entries(r).slice(0,2);
          const tr=document.createElement('tr');
          tr.innerHTML = `${esc((first[0]||['',''])[1])}${esc((first[1]||['',''])[1])}`;
          body.appendChild(tr);
        });
      }
      if(!body.children.length) body.innerHTML=`No analytics rows.`;
    }catch(e){
      body.innerHTML=`Failed to load analytics.`;
    }
  }

  /* ======= WALLET ======= */
  async function loadWallet(){
    const list=$('walletList'); list.innerHTML='';
    try{
      const j = await jget(api("wallet"));
      $('wallet-balance').textContent = usd(j.balance||0);
      (j.history||[]).forEach(tx=>{
        const div=document.createElement('div'); div.className='card span-6';
        div.innerHTML = `

            
${esc(tx.date||'')}
${esc(tx.method||'')}

            
${usd(tx.amount||0)}
${esc(tx.status||'COMPLETED')}

          
`;
        list.appendChild(div);
      });
      if(!list.children.length) list.innerHTML=`
No transactions.
`;
    }catch(e){
      list.innerHTML=`
Failed to load wallet.
`;
    }
  }

  /* ======= MONITORING ======= */
  async function loadMonitoring(){
    const body=$('#tbl-monitor tbody'); body.innerHTML='';
    try{
      const j = await jget(api("monitoring"));
      (j.events||[]).forEach(ev=>{
        const tr=document.createElement('tr');
        tr.innerHTML=`${esc(ev.time||'')}${esc(ev.type||'')}${esc(ev.msg||'')}`;
        body.appendChild(tr);
      });
      if(!body.children.length) body.innerHTML=`No events.`;
    }catch(e){
      body.innerHTML=`Failed to load events.`;
    }
  }

  /* ======= SETTINGS ======= */
  async function loadSettings(){
    try{
      const j = await jget(api("config"));
      $('cfgDump').textContent = JSON.stringify(j,null,2);
    }catch(e){
      $('cfgDump').textContent = 'Failed to load config.';
    }
  }

  /* ======= INIT ======= */
  window.addEventListener('DOMContentLoaded', ()=>{
    const h=(location.hash||'#dashboard').replace('#','');
    showTab(h);
  });
  window.addEventListener('hashchange', ()=>{
    const h=(location.hash||'#dashboard').replace('#','');
    showTab(h);
  });
